$$
Cn \doteq C{n-1}+W_n
$$

$$
\begin{align}
V{n+1} &\doteq \frac{\sum{k=1}^n W_k Gk}{\sum{k=1}^n Wk} \
&= \frac{\sum{k=1}^n W_k G_k}{C_n} \
&=\frac{W_nGn+\sum{k=1}^{n-1} W_k G_k}{C_n} \
&={W_nG_n \over Cn}+{C{n-1} \over Cn}\frac{\sum{k=1}^{n-1} W_k Gk}{C{n-1}} \
&= {W_nG_n \over Cn} + {C{n-1} \over C_n}V_n \
&= {W_nG_n \over C_n} + {C_n-W_n \over C_n}V_n \
&=V_n + {W_n \over C_n}\left( G_n - V_n \right)
\end{align}
